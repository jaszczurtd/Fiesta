cmake_minimum_required(VERSION 3.20)

project(FiestaArduinoFirmware NONE)

set(FIESTA_PROJECT_DIR "" CACHE PATH "Fiesta firmware module directory")
set(ARDUINO_CLI "arduino-cli" CACHE STRING "arduino-cli executable")
set(ARDUINO_FQBN "rp2040:rp2040:rpipico" CACHE STRING "Arduino fully qualified board name")
set(ARDUINO_SKETCHBOOK "" CACHE PATH "Compatibility cache var accepted from jh-vscode")
set(ARDUINO_UPLOAD_PORT "" CACHE STRING "Arduino upload port")
set(ARDUINO_VERBOSE OFF CACHE BOOL "Enable verbose Arduino CLI output")
set(JH_ROOT "" CACHE PATH "Compatibility cache var accepted from jh-vscode")
set(JH_USB_MANUFACTURER "" CACHE STRING "Compatibility cache var accepted from jh-vscode")
set(JH_USB_PRODUCT "" CACHE STRING "Compatibility cache var accepted from jh-vscode")
set(FIESTA_WARNINGS OFF CACHE BOOL "Enable Arduino warnings")
set(FIESTA_WERROR OFF CACHE BOOL "Treat compiler warnings as errors")
set(FIESTA_LIBRARIES_DIR "" CACHE PATH "Arduino libraries directory")
set(FIESTA_USB_MANUFACTURER "Jaszczur" CACHE STRING "USB manufacturer string")
set(FIESTA_USB_PRODUCT "Fiesta Firmware" CACHE STRING "USB product string")

if(NOT FIESTA_PROJECT_DIR)
    message(FATAL_ERROR "FIESTA_PROJECT_DIR is required")
endif()

get_filename_component(FIESTA_PROJECT_DIR "${FIESTA_PROJECT_DIR}" ABSOLUTE)
get_filename_component(FIESTA_MODULE_NAME "${FIESTA_PROJECT_DIR}" NAME)

if(NOT IS_DIRECTORY "${FIESTA_PROJECT_DIR}")
    message(FATAL_ERROR "Fiesta module directory not found: ${FIESTA_PROJECT_DIR}")
endif()

set(APP_NAME "${FIESTA_MODULE_NAME}")
set(SKETCH_DIR "${CMAKE_BINARY_DIR}/sketch/${APP_NAME}")
set(ARDUINO_BUILD_DIR "${FIESTA_PROJECT_DIR}/.build/arduino")
set(FIESTA_OUTPUT_DIR "${FIESTA_PROJECT_DIR}/.build")
set(FIRMWARE_BASENAME "${FIESTA_MODULE_NAME}")
set(FIRMWARE_ELF "${FIESTA_OUTPUT_DIR}/${FIRMWARE_BASENAME}.elf")
set(FIRMWARE_BIN "${FIESTA_OUTPUT_DIR}/${FIRMWARE_BASENAME}.bin")
set(FIRMWARE_UF2 "${FIESTA_OUTPUT_DIR}/${FIRMWARE_BASENAME}.uf2")
set(FIRMWARE_MAP "${FIESTA_OUTPUT_DIR}/${FIRMWARE_BASENAME}.map")
set(LEGACY_FIRMWARE_ELF "${FIESTA_OUTPUT_DIR}/firmware.elf")
set(LEGACY_FIRMWARE_BIN "${FIESTA_OUTPUT_DIR}/firmware.bin")
set(LEGACY_FIRMWARE_UF2 "${FIESTA_OUTPUT_DIR}/firmware.uf2")
set(LEGACY_FIRMWARE_MAP "${FIESTA_OUTPUT_DIR}/firmware.map")
set(LEGACY_FIRMWARE_MANIFEST "${FIESTA_OUTPUT_DIR}/firmware.manifest.json")
set(COMPILE_COMMANDS_OUT "${FIESTA_OUTPUT_DIR}/compile_commands.json")
set(FIRMWARE_ENTRY_HEADER "${FIESTA_PROJECT_DIR}/firmware_entry.h")
get_filename_component(FIESTA_COMMON_DIR "${CMAKE_CURRENT_LIST_DIR}/../.." ABSOLUTE)
set(FIESTA_UPLOAD_BUILT_UF2_SCRIPT "${FIESTA_COMMON_DIR}/scripts/fiesta-upload-built-uf2.sh")

find_program(BASH_EXECUTABLE bash REQUIRED)

file(REMOVE_RECURSE "${SKETCH_DIR}")
file(MAKE_DIRECTORY "${SKETCH_DIR}" "${ARDUINO_BUILD_DIR}" "${FIESTA_OUTPUT_DIR}")

if(NOT EXISTS "${FIRMWARE_ENTRY_HEADER}")
    message(FATAL_ERROR "Missing Fiesta firmware entry header: ${FIRMWARE_ENTRY_HEADER}")
endif()

set(_entry_code [=[
// Auto-generated by CMake - do not edit or commit.
// Real application code lives in this Fiesta module directory.
#include "firmware_entry.h"

void setup(void) {
    initialization();
}

void loop(void) {
    looper();
}

#if defined(FIESTA_ENABLE_CORE1) && FIESTA_ENABLE_CORE1
void setup1(void) {
    initialization1();
}

void loop1(void) {
    looper1();
}
#endif
]=])

file(WRITE "${SKETCH_DIR}/${APP_NAME}.ino" "${_entry_code}")

file(GLOB FIESTA_SKETCH_SOURCES CONFIGURE_DEPENDS
    "${FIESTA_PROJECT_DIR}/*.c"
    "${FIESTA_PROJECT_DIR}/*.cpp"
)

foreach(_src IN LISTS FIESTA_SKETCH_SOURCES)
    get_filename_component(_src_name "${_src}" NAME)
    file(CREATE_LINK "${_src}" "${SKETCH_DIR}/${_src_name}" SYMBOLIC COPY_ON_ERROR)
endforeach()

set(_extra_flags "-I${FIESTA_PROJECT_DIR}")
if(FIESTA_WERROR)
    string(APPEND _extra_flags " -Werror")
endif()

set(ARDUINO_LIBRARY_ARGS)
if(FIESTA_LIBRARIES_DIR)
    list(APPEND ARDUINO_LIBRARY_ARGS --libraries "${FIESTA_LIBRARIES_DIR}")
endif()

set(ARDUINO_COMMON_ARGS
    compile
    --fqbn "${ARDUINO_FQBN}"
    --build-path "${ARDUINO_BUILD_DIR}"
    ${ARDUINO_LIBRARY_ARGS}
    --build-property "compiler.cpp.extra_flags=${_extra_flags}"
    --build-property "compiler.c.extra_flags=${_extra_flags}"
    --build-property "build.usb_manufacturer=\"${FIESTA_USB_MANUFACTURER}\""
    --build-property "build.usb_product=\"${FIESTA_USB_PRODUCT}\""
)

if(FIESTA_WARNINGS)
    list(APPEND ARDUINO_COMMON_ARGS --warnings all)
endif()

if(ARDUINO_VERBOSE)
    list(APPEND ARDUINO_COMMON_ARGS --verbose)
endif()

function(fiesta_add_arduino_target target_name)
    set(options DEBUG UPLOAD COMPILE_DB)
    cmake_parse_arguments(ARG "${options}" "" "" ${ARGN})

    set(_cmd "${ARDUINO_CLI}" ${ARDUINO_COMMON_ARGS})

    if(ARG_DEBUG)
        list(APPEND _cmd --optimize-for-debug)
    endif()

    if(ARG_COMPILE_DB)
        list(APPEND _cmd --only-compilation-database)
    endif()

    if(ARG_UPLOAD)
        list(APPEND _cmd --upload)
        if(ARDUINO_UPLOAD_PORT)
            list(APPEND _cmd --port "${ARDUINO_UPLOAD_PORT}")
        endif()
    endif()

    list(APPEND _cmd "${SKETCH_DIR}")

    set(_cleanup_outputs "${COMPILE_COMMANDS_OUT}")
    if(NOT ARG_COMPILE_DB)
        list(APPEND _cleanup_outputs
            "${FIRMWARE_ELF}"
            "${FIRMWARE_BIN}"
            "${FIRMWARE_UF2}"
            "${FIRMWARE_MAP}"
            "${LEGACY_FIRMWARE_ELF}"
            "${LEGACY_FIRMWARE_BIN}"
            "${LEGACY_FIRMWARE_UF2}"
            "${LEGACY_FIRMWARE_MAP}"
            "${LEGACY_FIRMWARE_MANIFEST}"
        )
    endif()

    add_custom_target("${target_name}"
        COMMAND "${CMAKE_COMMAND}" -E rm -rf "${ARDUINO_BUILD_DIR}"
        COMMAND "${CMAKE_COMMAND}" -E make_directory "${ARDUINO_BUILD_DIR}"
        COMMAND "${CMAKE_COMMAND}" -E rm -f ${_cleanup_outputs}
        COMMAND ${_cmd}
        WORKING_DIRECTORY "${FIESTA_PROJECT_DIR}"
        USES_TERMINAL
        VERBATIM
    )

    if(ARG_COMPILE_DB)
        add_custom_command(TARGET "${target_name}" POST_BUILD
            COMMAND "${CMAKE_COMMAND}" -E copy_if_different
                "${ARDUINO_BUILD_DIR}/compile_commands.json"
                "${COMPILE_COMMANDS_OUT}"
            VERBATIM
        )
    elseif(NOT ARG_UPLOAD)
        add_custom_command(TARGET "${target_name}" POST_BUILD
            COMMAND "${CMAKE_COMMAND}" -E copy
                "${ARDUINO_BUILD_DIR}/${APP_NAME}.ino.elf"
                "${FIRMWARE_ELF}"
            COMMAND "${CMAKE_COMMAND}" -E copy
                "${ARDUINO_BUILD_DIR}/${APP_NAME}.ino.bin"
                "${FIRMWARE_BIN}"
            COMMAND "${CMAKE_COMMAND}" -E copy
                "${ARDUINO_BUILD_DIR}/${APP_NAME}.ino.uf2"
                "${FIRMWARE_UF2}"
            COMMAND "${CMAKE_COMMAND}" -E copy
                "${ARDUINO_BUILD_DIR}/${APP_NAME}.ino.map"
                "${FIRMWARE_MAP}"
            COMMAND "${CMAKE_COMMAND}" -E copy
                "${ARDUINO_BUILD_DIR}/${APP_NAME}.ino.elf"
                "${LEGACY_FIRMWARE_ELF}"
            COMMAND "${CMAKE_COMMAND}" -E copy
                "${ARDUINO_BUILD_DIR}/${APP_NAME}.ino.bin"
                "${LEGACY_FIRMWARE_BIN}"
            COMMAND "${CMAKE_COMMAND}" -E copy
                "${ARDUINO_BUILD_DIR}/${APP_NAME}.ino.uf2"
                "${LEGACY_FIRMWARE_UF2}"
            COMMAND "${CMAKE_COMMAND}" -E copy
                "${ARDUINO_BUILD_DIR}/${APP_NAME}.ino.map"
                "${LEGACY_FIRMWARE_MAP}"
            VERBATIM
        )
    endif()
endfunction()

fiesta_add_arduino_target(firmware)
fiesta_add_arduino_target(firmware_debug DEBUG)
fiesta_add_arduino_target(firmware_compile_db COMPILE_DB)

set(_arduino_verbose_flag 0)
if(ARDUINO_VERBOSE)
    set(_arduino_verbose_flag 1)
endif()

add_custom_target(firmware_upload
    COMMAND "${BASH_EXECUTABLE}"
        "${FIESTA_UPLOAD_BUILT_UF2_SCRIPT}"
        "${FIESTA_PROJECT_DIR}"
        "${LEGACY_FIRMWARE_UF2}"
        "${ARDUINO_UPLOAD_PORT}"
        "${_arduino_verbose_flag}"
    DEPENDS firmware
    WORKING_DIRECTORY "${FIESTA_PROJECT_DIR}"
    USES_TERMINAL
    VERBATIM
)

message(STATUS "Fiesta module: ${FIESTA_MODULE_NAME}")
message(STATUS "Arduino FQBN: ${ARDUINO_FQBN}")
message(STATUS "Generated sketch: ${SKETCH_DIR}")
message(STATUS "Arduino build dir: ${ARDUINO_BUILD_DIR}")
